次方求模模板

就是求(a^b)% p其中 a, b, p 数值比较大 取余公式：(ab)% p = (a%p b%p)% p 

//次方求模模板
template<class Type>
inline Type ModPow(Type m,Type n,Type p) //m的n次方模p
{
    if(n==0) return 1;
    if (n==1) return m%p;
        Type tmp=ModPow(m,n>>1,p);
    return (tmp*tmp%p)*((n%2?m:1)%p)%p;
}

哈希模版(hash template)

const int  MAX=1000003;
template <class T>
class hash
{
private:
    int pos;
    int next[MAX];
    int head[MAX];
    T key[MAX];
public:
    hash();
    bool search(T x);
    void push(T x);
};
template <class T>
hash<T>::hash()
{
    pos=0;
    memset(next,-1,sizeof(next));
    memset(head,-1,sizeof(head));
    //memset(key,-1,sizeof(key));
}
template <class T>
inline bool hash<T>::search(const T x)
{
    int temp=x%MAX;
    int t=head[temp];
    while(t!=-1)
    {
        if (key[t]==x)
        {
            return 1;
        }
        t=next[t];
    }
    return 0;
}
template <class T>
inline void hash<T>::push(const T x)
{
    int temp=x% MAX;
    if (head[temp]!=-1)
    {
        next[pos]=head[temp];
    }
    head[temp]=pos;
    key[pos]=x;
    pos++;
}
